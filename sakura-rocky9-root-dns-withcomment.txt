#!/bin/bash
# Claude.ai MCP サーバー構築 for さくらのVPS (RockyLinux 9)
# パラメータ: DOMAIN / LETSENCRYPT_EMAIL / CLIENT_SECRET / NS1_IP
# このコードは Claude Sonnet 4.6 が書きました。
#
# 概要:
#   1. swapfile 1GB
#   2. パッケージ（nginx/bind/certbot/postfix/dnf-automatic）
#   3. Node.js 20
#   4. MCPサーバー (exec_command/nginx_reload)
#      OAuth2簡易認証: /tokenエンドポイントで初回のみtoken発行
#      tokenは/etc/mcp-server/tokenに保存・アクセスのたびに読み込み
#      rm /etc/mcp-server/token で即座に再認証が可能
#   5. BIND プライマリDNS（allow-recursionでオープンリゾルバ防止）
#   6. npm install
#   7. systemdサービス（EnvironmentFileでシークレットを隔離）
#   8. Nginx（MCPpathはaccess_log off・/tokenエンドポイントを公開）
#   9. SSL: DNS解決を最大30分待機しcertbot実行
#  10. firewalld
#  11. 週次reboot（日曜3時・Persistent=false）
#  12. スタートアップログをメール送信
set -euo pipefail

DOMAIN=@@@DOMAIN@@@
LETSENCRYPT_EMAIL=@@@LETSENCRYPT_EMAIL@@@
CLIENT_SECRET=@@@CLIENT_SECRET@@@
MCP_PORT=3000
NS1_IP=@@@NS1_IP@@@

exec > >(tee -a /var/log/mcp-startup.log) 2>&1
echo "===== $(date) ====="

# 1. swapfile
fallocate -l 1G /swapfile
chmod og-rwx /swapfile
mkswap /swapfile
swapon /swapfile
echo '/swapfile none swap sw 0 0' | tee -a /etc/fstab

# 2. パッケージ
dnf update -y
dnf config-manager --enable crb
dnf install -y curl nginx epel-release bind-utils bind
dnf install -y certbot python3-certbot-nginx
dnf install -y postfix s-nail dnf-automatic
postconf -e 'inet_protocols = ipv4'  # IPv6なしVPSでの起動エラー対策
systemctl enable postfix && systemctl start postfix
sed -i 's/upgrade_type = default/upgrade_type = security/' /etc/dnf/automatic.conf
sed -i 's/apply_updates = no/apply_updates = yes/' /etc/dnf/automatic.conf
systemctl enable --now dnf-automatic-install.timer  # セキュリティアップデート自動化

# 3. Node.js 20
curl -fsSL https://rpm.nodesource.com/setup_20.x | bash -
dnf install -y nodejs

# 4. MCPサーバー
MCP_DIR="/opt/mcp-server"
mkdir -p "$MCP_DIR"

cat > "$MCP_DIR/package.json" << 'EOF'
{
  "name": "vps-mcp-server",
  "version": "1.0.0",
  "type": "module",
  "dependencies": {
    "@modelcontextprotocol/sdk": "^1.0.0",
    "express": "^4.18.0",
    "zod": "^3.22.0"
  }
}
EOF

cat > "$MCP_DIR/index.mjs" << 'MCPEOF'
import express from "express";
import { McpServer } from "@modelcontextprotocol/sdk/server/mcp.js";
import { SSEServerTransport } from "@modelcontextprotocol/sdk/server/sse.js";
import { z } from "zod";
import { exec } from "child_process";  // 非同期: イベントループをブロックしない
import { readFileSync, writeFileSync } from "fs";
import crypto from "crypto";

const SECRET_FILE = "/etc/mcp-server/secret";
const TOKEN_FILE  = "/etc/mcp-server/token";
const PORT = parseInt(process.env.MCP_PORT || "3000");

const app = express();
app.use(express.json());

// tokenファイルを都度読み込む（rm tokenで即座に再認証が可能）
function getToken() { try { return readFileSync(TOKEN_FILE, "utf8").trim(); } catch { return null; } }

// /token: 初回のみtoken発行・以降はgetToken()がtruthyで即座に403
app.post("/token", (req, res) => {
  let secret; try { secret = readFileSync(SECRET_FILE, "utf8").trim(); } catch { secret = null; }
  if (getToken() || req.body.client_secret !== secret) return res.status(403).json({ error: "Forbidden" });
  const token = crypto.randomUUID();
  writeFileSync(TOKEN_FILE, token, { mode: 0o600 });
  res.json({ access_token: token, token_type: "Bearer" });
});

// Bearer token認証
function auth(req, res, next) {
  const t = (req.headers.authorization || "").replace("Bearer ", "");
  const s = getToken();
  if (!s || t !== s) return res.status(401).json({ error: "Unauthorized" });
  next();
}

// 共通のPromiseラッパー
function run(cmd, timeout = 30000) {
  return new Promise((resolve, reject) => {
    exec(cmd, { timeout, maxBuffer: 1024*1024 }, (err, stdout, stderr) => {
      if (err) { err.stdout = stdout; err.stderr = stderr; reject(err); }
      else resolve(stdout);
    });
  });
}

app.get("/health", (_req, res) => res.json({ status: "ok" }));

function createMcpServer() {
  const server = new McpServer({ name: "sakura-vps", version: "1.0.0" });

  // ツール1: 任意コマンド実行
  server.tool(
    "exec_command",
    "VPS上でシェルコマンドを実行（root権限）。nginx reload/restartはnginx_reloadツールを使うこと。",
    { command: z.string() },
    async ({ command }) => {
      try {
        const out = await run(command);
        return { content: [{ type: "text", text: out || "(出力なし)" }] };
      } catch (e) {
        const isTimeout = e.killed || e.code === "ETIMEDOUT";
        const msg = [
          isTimeout ? "タイムアウト（30秒超過）" : e.message,
          e.stdout ? "stdout: " + e.stdout : "",
          e.stderr ? "stderr: " + e.stderr : "",
        ].filter(Boolean).join("\n");
        return { content: [{ type: "text", text: "エラー: " + msg }], isError: true };
      }
    }
  );

  // ツール2: Nginxリロード専用（SSE切断を事前応答してから実行）
  server.tool(
    "nginx_reload",
    "Nginxをリロードする専用ツール。nginx reload/restartは必ずこれを使うこと、2秒後SSEセッションが切断されるので再接続が必要。",
    {},
    async () => {
      try {
        await run("nginx -t");  // 設定テストを先に実行
      } catch (e) {
        return { content: [{ type: "text", text: "設定テスト失敗:\n" + (e.stderr || e.message) }], isError: true };
      }
      setTimeout(() => run("nginx -s reload"), 2000);  // 2秒後に実行
      return { content: [{ type: "text", text: "nginx設定テストOK、2秒後reload。SSEセッションが切断されるので再接続してください。" }] };
    }
  );

  return server;
}

const transports = new Map();

app.get("/mcp/sse", auth, async (req, res) => {
  res.setHeader("Content-Type", "text/event-stream");
  res.setHeader("Cache-Control", "no-cache");
  res.setHeader("Connection", "keep-alive");
  const server = createMcpServer();
  const transport = new SSEServerTransport("/mcp/messages", res);
  transports.set(transport.sessionId, transport);
  await server.connect(transport);
  req.on("close", () => {
    transports.delete(transport.sessionId);
    transport.close();
  });
});

app.post("/mcp/messages", auth, async (req, res) => {
  const transport = transports.get(req.query.sessionId);
  if (!transport) return res.status(404).json({ error: "Session not found" });
  await transport.handlePostMessage(req, res, req.body);
});

app.listen(PORT, "127.0.0.1", () => console.log(`MCP listening on ${PORT}`));
MCPEOF

# 5. BIND (プライマリDNS)
SERIAL=$(date +%Y%m%d01)
mkdir -p /etc/named

cat > /var/named/${DOMAIN}.zone << ZONEOF
\$TTL 3600
@ IN SOA ns1.${DOMAIN}. admin.${DOMAIN}. (
    ${SERIAL} 3600 900 604800 300 )
@ IN NS ns1.${DOMAIN}.
@ IN NS ns1.dns.ne.jp.
@ IN NS ns2.dns.ne.jp.
@ IN A  ${NS1_IP}
ns1 IN A ${NS1_IP}
ZONEOF
chown named:named /var/named/${DOMAIN}.zone

cat > /etc/named/named.conf.local << NAMEDEOF
zone "${DOMAIN}" {
    type master;
    file "/var/named/${DOMAIN}.zone";
    allow-transfer { 210.188.224.9; 210.224.172.13; };  # さくらゾーン転送元IP
    also-notify   { 61.211.236.1; 133.167.21.1; };      # ns1/ns2.dns.ne.jp
};
NAMEDEOF

grep -q 'named.conf.local' /etc/named.conf \
  || echo 'include "/etc/named/named.conf.local";' >> /etc/named.conf

sed -i 's/listen-on port 53 { 127.0.0.1; };/listen-on port 53 { any; };/' /etc/named.conf
grep -q 'listen-on port 53' /etc/named.conf \
  || sed -i '/^options/a\\tlisten-on port 53 { any; };' /etc/named.conf

sed -i 's/allow-query\s*{\s*localhost;\s*};/allow-query { any; };/' /etc/named.conf
grep -q 'allow-query' /etc/named.conf \
  || sed -i '/^options/a\\tallow-query { any; };' /etc/named.conf

# allow-recursion: ローカルのみに限定しオープンリゾルバを防ぐ
grep -q 'allow-recursion' /etc/named.conf \
  || sed -i '/^options/a\\tallow-recursion { 127.0.0.1; };' /etc/named.conf

systemctl enable named && systemctl start named

sleep 10
rndc querylog on
rndc notify "${DOMAIN}"
journalctl -u named -n 20 | grep -i "xfer\|transfer\|axfr\|notify" || true

# 6. npm install
cd "$MCP_DIR" && npm install --production

# 7. systemdサービス
# EnvironmentFileを使いシークレットを隔離（systemctl catで表示されない）
cat > /etc/systemd/system/mcp-server.service << EOF
[Unit]
Description=Claude.ai MCP Server
After=network.target
[Service]
Type=simple
User=root
WorkingDirectory=${MCP_DIR}
ExecStart=/usr/bin/node ${MCP_DIR}/index.mjs
Restart=always
RestartSec=5
EnvironmentFile=/etc/mcp-server/env
StandardOutput=append:/var/log/mcp-server.log
StandardError=append:/var/log/mcp-server.log
[Install]
WantedBy=multi-user.target
EOF

mkdir -p /etc/mcp-server
echo "$CLIENT_SECRET" > /etc/mcp-server/secret
chmod 600 /etc/mcp-server/secret
cat > /etc/mcp-server/env << EOF
MCP_PORT=${MCP_PORT}
EOF
chmod 600 /etc/mcp-server/env
systemctl daemon-reload && systemctl enable mcp-server && systemctl start mcp-server

# 8. Nginx
# log_formatはhttpブロックに書く必要がある（server blockに書くと他のconf.dから参照不可）
cat > /etc/nginx/conf.d/mcp.conf << EOF
server {
    listen 80;
    server_name ${DOMAIN};
    location /.well-known/acme-challenge/ { root /usr/share/nginx/html; }
    location ~ ^/mcp/(sse|messages) {
        access_log off;  # APIkeyがログに残らないようオフ
        proxy_pass http://127.0.0.1:${MCP_PORT};
        proxy_http_version 1.1;
        proxy_buffering off;
        proxy_set_header Connection '';
        proxy_read_timeout 3600s;
    }
    location /token  { proxy_pass http://127.0.0.1:${MCP_PORT}/token; }
    location /health { proxy_pass http://127.0.0.1:${MCP_PORT}/health; }
    location / { root /usr/share/nginx/html; index index.html; try_files \$uri \$uri/ =404; }
}
EOF
systemctl enable nginx && systemctl start nginx
nginx -t && systemctl reload nginx

# 9. SSL (HTTP-01方式。DNS解決を最大30分待機しcertbot実行)
mkdir -p /etc/letsencrypt/renewal-hooks/deploy
cat > /etc/letsencrypt/renewal-hooks/deploy/reload-nginx.sh << 'HOOKEOF'
#!/bin/bash
systemctl reload nginx  # 証明書更新後にNginxに反映
HOOKEOF
chmod +x /etc/letsencrypt/renewal-hooks/deploy/reload-nginx.sh

VPS_IP=$(curl -s ifconfig.me 2>/dev/null || echo "")
echo "DNS解決待機中..."
DOMAIN_IP=""
for i in $(seq 1 30); do
    DOMAIN_IP=$(dig +short "$DOMAIN" | tail -1)
    if [ -n "$DOMAIN_IP" ] && [ "$VPS_IP" = "$DOMAIN_IP" ]; then
        echo "DNS解決（${i}回目）"
        break
    fi
    echo "${i}: VPS=${VPS_IP} DNS=${DOMAIN_IP}"
    sleep 60
done

if [ -n "$DOMAIN_IP" ] && [ "$VPS_IP" = "$DOMAIN_IP" ]; then
    certbot --nginx --non-interactive --agree-tos \
        --email "$LETSENCRYPT_EMAIL" -d "$DOMAIN" \
        && echo "SSL完了" \
        || echo "SSL失敗"
else
    echo "DNS未解決のためSSLスキップ"
fi

# 10. firewalld
systemctl enable firewalld && systemctl start firewalld
for X in ssh http https; do firewall-cmd --permanent --add-service=$X; done
for X in 53/udp 53/tcp 3000/tcp; do firewall-cmd --permanent --add-port=$X; done
firewall-cmd --reload

# 11. 週次reboot (日曜3時・Persistent=false: 起動時スキップされる)
cat > /etc/systemd/system/weekly-reboot.timer << 'EOF'
[Unit]
Description=Weekly reboot
[Timer]
OnCalendar=Sun 03:00
Persistent=false
[Install]
WantedBy=timers.target
EOF
cat > /etc/systemd/system/weekly-reboot.service << 'EOF'
[Unit]
Description=Weekly reboot
[Service]
Type=oneshot
ExecStart=/sbin/reboot
EOF
systemctl enable --now weekly-reboot.timer

# 12. スタートアップログをメール送信（certbot結果確認用）
VPS_IP=$(curl -s ifconfig.me 2>/dev/null || echo "失敗")
echo "===== $(date) ====="
echo "${VPS_IP} ${DOMAIN}"
echo "SSE:https://${DOMAIN}/mcp/sse TOKEN:https://${DOMAIN}/token"
tail -50 /var/log/mcp-startup.log \
    | mail -s "VPS startup: ${DOMAIN}" "$LETSENCRYPT_EMAIL"
