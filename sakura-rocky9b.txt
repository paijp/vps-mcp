#!/bin/bash
# Claude.ai MCP サーバー構築 for さくらのVPS (RockyLinux)
# パラメータ: DOMAIN / LETSENCRYPT_EMAIL / MCP_API_KEY / NS1_IP
set -euo pipefail

DOMAIN=@@@DOMAIN@@@
LETSENCRYPT_EMAIL=@@@LETSENCRYPT_EMAIL@@@
MCP_API_KEY=@@@MCP_API_KEY@@@
MCP_PORT=3000
NS1_IP=@@@NS1_IP@@@

exec > >(tee -a /var/log/mcp-startup.log) 2>&1
echo "===== Setup Start: $(date) ====="

# 1. swapfile
fallocate -l 1G /swapfile
chmod og-rwx /swapfile
mkswap /swapfile
swapon /swapfile
echo '/swapfile none swap sw 0 0' | tee -a /etc/fstab

# 2. パッケージ
dnf update -y
dnf config-manager --enable crb
dnf install -y curl nginx epel-release bind-utils bind
dnf install -y certbot python3-certbot-nginx python3-certbot-dns-rfc2136

# 3. Node.js 20
curl -fsSL https://rpm.nodesource.com/setup_20.x | bash -
dnf install -y nodejs

# 4. MCPサーバー
MCP_DIR="/opt/mcp-server"
mkdir -p "$MCP_DIR"

cat > "$MCP_DIR/package.json" << 'EOF'
{
  "name": "vps-mcp-server",
  "version": "1.0.0",
  "type": "module",
  "dependencies": {
    "@modelcontextprotocol/sdk": "^1.0.0",
    "express": "^4.18.0",
    "zod": "^3.22.0"
  }
}
EOF

cat > "$MCP_DIR/index.mjs" << 'MCPEOF'
import express from "express";
import { McpServer } from "@modelcontextprotocol/sdk/server/mcp.js";
import { SSEServerTransport } from "@modelcontextprotocol/sdk/server/sse.js";
import { z } from "zod";
import { execSync } from "child_process";
import { readFileSync, writeFileSync } from "fs";

const API_KEY = process.env.MCP_API_KEY;
const PORT = parseInt(process.env.MCP_PORT || "3000");

const app = express();
app.use(express.json());

function auth(req, res, next) {
  const key = req.params.key || req.query.key || "";
  if (!API_KEY || key !== API_KEY) return res.status(401).json({ error: "Unauthorized" });
  next();
}

app.get("/health", (_req, res) => res.json({ status: "ok" }));

function createMcpServer() {
  const server = new McpServer({ name: "sakura-vps", version: "1.0.0" });

  server.tool(
    "exec_command",
    "VPS上でシェルコマンドを実行する（root権限）",
    { command: z.string() },
    async ({ command }) => {
      try {
        const out = execSync(command, { timeout: 30000, maxBuffer: 1024*1024, encoding: "utf8" });
        return { content: [{ type: "text", text: out || "(出力なし)" }] };
      } catch (e) {
        return { content: [{ type: "text", text: "エラー: " + e.message }], isError: true };
      }
    }
  );

  server.tool(
    "read_file",
    "VPS上のファイルを読み込む",
    { path: z.string() },
    async ({ path }) => {
      try {
        return { content: [{ type: "text", text: readFileSync(path, "utf8") }] };
      } catch (e) {
        return { content: [{ type: "text", text: "エラー: " + e.message }], isError: true };
      }
    }
  );

  server.tool(
    "write_file",
    "VPS上のファイルに内容を書き込む（上書き）",
    { path: z.string(), content: z.string() },
    async ({ path, content }) => {
      try {
        writeFileSync(path, content, "utf8");
        return { content: [{ type: "text", text: "書き込み完了: " + path }] };
      } catch (e) {
        return { content: [{ type: "text", text: "エラー: " + e.message }], isError: true };
      }
    }
  );

  return server;
}

const transports = new Map();

app.get("/mcp/:key/sse", auth, async (req, res) => {
  res.setHeader("Content-Type", "text/event-stream");
  res.setHeader("Cache-Control", "no-cache");
  res.setHeader("Connection", "keep-alive");
  const server = createMcpServer();
  const transport = new SSEServerTransport(`/mcp/${req.params.key}/messages`, res);
  transports.set(transport.sessionId, transport);
  await server.connect(transport);
  req.on("close", () => {
    transports.delete(transport.sessionId);
    transport.close();
  });
});

app.post("/mcp/:key/messages", auth, async (req, res) => {
  const transport = transports.get(req.query.sessionId);
  if (!transport) return res.status(404).json({ error: "Session not found" });
  await transport.handlePostMessage(req, res, req.body);
});

app.listen(PORT, "127.0.0.1", () => console.log(`MCP listening on ${PORT}`));
MCPEOF

# 5. BIND (プライマリDNS)
SERIAL=$(date +%Y%m%d01)
mkdir -p /etc/named /etc/certbot

cat > /var/named/${DOMAIN}.zone << ZONEOF
\$TTL 3600
@ IN SOA ns1.${DOMAIN}. admin.${DOMAIN}. (
    ${SERIAL} 3600 900 604800 300 )
@ IN NS ns1.${DOMAIN}.
@ IN NS ns1.dns.ne.jp.
@ IN NS ns2.dns.ne.jp.
@ IN A  ${NS1_IP}
ns1 IN A ${NS1_IP}
ZONEOF
chown named:named /var/named/${DOMAIN}.zone

tsig-keygen -a hmac-sha256 certbot-key > /etc/named/certbot.key
TSIG_SECRET=$(grep secret /etc/named/certbot.key | awk '{print $2}' | tr -d '";')

cat > /etc/named/named.conf.local << NAMEDEOF
key "certbot-key" {
    algorithm hmac-sha256;
    secret "${TSIG_SECRET}";
};
zone "${DOMAIN}" {
    type master;
    file "/var/named/${DOMAIN}.zone";
    allow-transfer { 210.188.224.9; 210.224.172.13; }; // さくらゾーン転送元
    also-notify   { 61.211.236.1; 133.167.21.1; };    // ns1/ns2.dns.ne.jp
    update-policy { grant certbot-key name _acme-challenge.${DOMAIN}. TXT; };
};
NAMEDEOF

grep -q 'named.conf.local' /etc/named.conf \
  || echo 'include "/etc/named/named.conf.local";' >> /etc/named.conf

# listen-onをanyに変更（デフォルトは127.0.0.1のみで外部から到達不可）
sed -i 's/listen-on port 53 { 127.0.0.1; };/listen-on port 53 { any; };/' /etc/named.conf
grep -q 'listen-on port 53' /etc/named.conf \
  || sed -i '/^options/a\\\tlisten-on port 53 { any; };' /etc/named.conf

# allow-queryをanyに変更（デフォルトはlocalhostのみで外部からの問い合わせ不可）
sed -i 's/allow-query\s*{\s*localhost;\s*};/allow-query { any; };/' /etc/named.conf
grep -q 'allow-query' /etc/named.conf \
  || sed -i '/^options/a\\\tallow-query { any; };' /etc/named.conf

systemctl enable named && systemctl start named

cat > /etc/certbot/rfc2136.ini << CREDEOF
dns_rfc2136_server = 127.0.0.1
dns_rfc2136_port = 53
dns_rfc2136_name = certbot-key
dns_rfc2136_secret = ${TSIG_SECRET}
dns_rfc2136_algorithm = HMAC-SHA256
CREDEOF
chmod 600 /etc/certbot/rfc2136.ini

# 6. npm install
cd "$MCP_DIR" && npm install --production

# 7. systemd
cat > /etc/systemd/system/mcp-server.service << EOF
[Unit]
Description=Claude.ai MCP Server
After=network.target
[Service]
Type=simple
User=root
WorkingDirectory=${MCP_DIR}
ExecStart=/usr/bin/node ${MCP_DIR}/index.mjs
Restart=always
RestartSec=5
Environment=MCP_API_KEY=${MCP_API_KEY}
Environment=MCP_PORT=${MCP_PORT}
StandardOutput=append:/var/log/mcp-server.log
StandardError=append:/var/log/mcp-server.log
[Install]
WantedBy=multi-user.target
EOF
systemctl daemon-reload && systemctl enable mcp-server && systemctl start mcp-server

# 8. Nginx
cat > /etc/nginx/conf.d/mcp.conf << EOF
log_format mcp_masked '\$remote_addr - [\$time_local] "\$request_method \$uri" \$status';

server {
    listen 80;
    server_name ${DOMAIN};
    location /.well-known/acme-challenge/ { root /usr/share/nginx/html; }
    location ~ ^/mcp/[^/]+/(sse|messages) {
        access_log /var/log/nginx/mcp_access.log mcp_masked;
        proxy_pass http://127.0.0.1:${MCP_PORT};
        proxy_http_version 1.1;
        proxy_buffering off;
        proxy_set_header Connection '';
        proxy_read_timeout 3600s;
    }
    location /health { proxy_pass http://127.0.0.1:${MCP_PORT}/health; }
    location / { root /usr/share/nginx/html; index index.html; try_files \$uri \$uri/ =404; }
}
EOF
systemctl enable nginx && systemctl start nginx
nginx -t && systemctl reload nginx

# 9. SSL
mkdir -p /etc/letsencrypt/renewal-hooks/deploy
cat > /etc/letsencrypt/renewal-hooks/deploy/reload-nginx.sh << 'HOOKEOF'
#!/bin/bash
systemctl reload nginx
HOOKEOF
chmod +x /etc/letsencrypt/renewal-hooks/deploy/reload-nginx.sh

VPS_IP=$(curl -s ifconfig.me 2>/dev/null || echo "")
DOMAIN_IP=$(dig +short "$DOMAIN" | tail -1)
if [ -n "$DOMAIN_IP" ] && [ "$VPS_IP" = "$DOMAIN_IP" ]; then
    certbot certonly --dns-rfc2136 \
        --dns-rfc2136-credentials /etc/certbot/rfc2136.ini \
        --non-interactive --agree-tos --email "$LETSENCRYPT_EMAIL" \
        -d "$DOMAIN" && echo "SSL取得完了" \
    || echo "SSL取得失敗 → 後で手動実行してください"
else
    echo "DNS未一致のためSSLスキップ（VPS:${VPS_IP} / DNS:${DOMAIN_IP}）"
    echo "一致後に実行: certbot certonly --dns-rfc2136 --dns-rfc2136-credentials /etc/certbot/rfc2136.ini --non-interactive --agree-tos --email ${LETSENCRYPT_EMAIL} -d ${DOMAIN}"
fi

# 10. firewalld
systemctl enable firewalld && systemctl start firewalld
firewall-cmd --permanent --add-service=ssh
firewall-cmd --permanent --add-service=http
firewall-cmd --permanent --add-service=https
firewall-cmd --permanent --add-port=53/udp
firewall-cmd --permanent --add-port=53/tcp
firewall-cmd --permanent --add-port=3000/tcp
firewall-cmd --reload

VPS_IP=$(curl -s ifconfig.me 2>/dev/null || echo "取得失敗")
echo "===== 完了: $(date) ====="
echo "IP: ${VPS_IP} / DOMAIN: ${DOMAIN}"
echo "claude.ai設定: URL=https://${DOMAIN}/mcp/${MCP_API_KEY}/sse"
echo "さくらDNS設定: コントロールパネル → セカンダリDNS → プライマリ: ${NS1_IP}"
echo "グルーレコード: ns1.${DOMAIN} → ${NS1_IP} をレジストラに登録してください"
